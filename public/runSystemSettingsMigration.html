END IF; END $$; -- Create a key-value store table for system settings CREATE TABLE IF NOT EXISTS system_settings ( key VARCHAR(100) PRIMARY KEY, value TEXT NOT NULL, description TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Insert default commission rate INSERT INTO system_settings (key, value, description) VALUES ('commission_rate', '0.50', 'Commission rate per kg in USD') ON CONFLICT (key) DO NOTHING; -- Migrate sample delivery address from old table if it exists DO $$ BEGIN IF EXISTS (SELECT 1 FROM information_schema.tables WHERE table_name = 'system_settings_old') THEN INSERT INTO system_settings (key, value, description) SELECT 'sample_delivery_address', sample_delivery_address, 'Default address for sample deliveries' FROM system_settings_old WHERE sample_delivery_address IS NOT NULL LIMIT 1 ON CONFLICT (key) DO NOTHING; END IF; END $$; -- Create function to update updated_at timestamp CREATE OR REPLACE FUNCTION update_system_settings_updated_at() RETURNS TRIGGER AS $$ BEGIN NEW.updated_at = CURRENT_TIMESTAMP; RETURN NEW; END; $$ LANGUAGE plpgsql; -- Create trigger for updating updated_at DROP TRIGGER IF EXISTS trigger_update_system_settings_updated_at ON system_settings; CREATE TRIGGER trigger_update_system_settings_updated_at BEFORE UPDATE ON system_settings FOR EACH ROW EXECUTE FUNCTION update_system_settings_updated_at();
