Run Transfer Migration

Copy and paste the migration SQL:

-- Transfer documents from destination to documents column WITH documents_to_transfer AS ( SELECT id, destination, documents, CASE -- Check if destination is an array of documents WHEN jsonb_typeof(destination) = 'array' AND destination::text LIKE '%"url"%' AND destination::text LIKE '%"name"%' THEN destination -- Check if destination is a single document object WHEN jsonb_typeof(destination) = 'object' AND (destination ? 'url' OR destination ? 'name' OR destination ? 'type') AND NOT (destination ? 'fbaWarehouse' OR destination ? 'address') THEN jsonb_build_array(destination) ELSE NULL END as docs_from_destination FROM shipments WHERE destination IS NOT NULL ) UPDATE shipments s SET documents = COALESCE( CASE WHEN s.documents IS NULL OR s.documents = '[]'::jsonb THEN dt.docs_from_destination ELSE s.documents || dt.docs_from_destination END, s.documents ), destination = NULL, updated_at = NOW() FROM documents_to_transfer dt WHERE s.id = dt.id AND dt.docs_from_destination IS NOT NULL; -- Show results SELECT COUNT(*) FILTER (WHERE documents IS NOT NULL AND documents != '[]'::jsonb) as shipments_with_documents, COUNT(*) FILTER (WHERE destination IS NOT NULL) as shipments_still_with_destination FROM shipments;
Verify Results
-- Verify transferred documents SELECT id, jsonb_array_length(COALESCE(documents, '[]'::jsonb)) as document_count, documents FROM shipments WHERE documents IS NOT NULL AND documents != '[]'::jsonb ORDER BY updated_at DESC LIMIT 10;

Method 2: Supabase Database Functions

Create a Reusable Database Function

Create this function once in your Supabase SQL Editor:

-- Create a function to transfer documents CREATE OR REPLACE FUNCTION transfer_documents_from_destination() RETURNS TABLE( transferred_count integer, total_documents integer ) LANGUAGE plpgsql AS $$ DECLARE transfer_count integer := 0; doc_count integer := 0; BEGIN -- Perform the transfer WITH documents_to_transfer AS ( SELECT id, destination, documents, CASE WHEN jsonb_typeof(destination) = 'array' AND destination::text LIKE '%"url"%' THEN destination WHEN jsonb_typeof(destination) = 'object' AND (destination ? 'url' OR destination ? 'name') AND NOT (destination ? 'fbaWarehouse') THEN jsonb_build_array(destination) ELSE NULL END as docs_from_destination FROM shipments WHERE destination IS NOT NULL ) UPDATE shipments s SET documents = COALESCE( CASE WHEN s.documents IS NULL OR s.documents = '[]'::jsonb THEN dt.docs_from_destination ELSE s.documents || dt.docs_from_destination END, s.documents ), destination = NULL, updated_at = NOW() FROM documents_to_transfer dt WHERE s.id = dt.id AND dt.docs_from_destination IS NOT NULL; -- Get counts GET DIAGNOSTICS transfer_count = ROW_COUNT; SELECT COUNT(*) INTO doc_count FROM shipments WHERE documents IS NOT NULL AND documents != '[]'::jsonb; RETURN QUERY SELECT transfer_count, doc_count; END; $$;

Then call it anytime with:

SELECT * FROM transfer_documents_from_destination();

const { createClient } = require('@supabase/supabase-js'); const supabaseUrl = 'https://jqcbgpwheckblkdejuwv.supabase.co'; const supabaseKey = 'your-service-role-key'; // Use service role key for admin operations const supabase = createClient(supabaseUrl, supabaseKey); async function transferDocuments() { console.log('Starting document transfer...'); // Fetch shipments with destination data const { data: shipments, error: fetchError } = await supabase .from('shipments') .select('*') .not('destination', 'is', null); if (fetchError) { console.error('Error fetching shipments:', fetchError); return; } console.log(`Found ${shipments.length} shipments with destination data`); let transferred = 0; for (const shipment of shipments) { const dest = shipment.destination; // Check if destination contains documents if (isDocumentData(dest)) { const docs = Array.isArray(dest) ? dest : [dest]; const allDocs = [...(shipment.documents || []), ...docs]; const { error: updateError } = await supabase .from('shipments') .update({ documents: allDocs, destination: null, updated_at: new Date().toISOString() }) .eq('id', shipment.id); if (!updateError) { transferred++; console.log(`‚úì Transferred ${docs.length} documents for shipment ${shipment.id}`); } else { console.error(`‚úó Error updating shipment ${shipment.id}:`, updateError); } } } console.log(`\nTransfer complete! Updated ${transferred} shipments`); } function isDocumentData(data) { if (!data || typeof data !== 'object') return false; const hasDocFields = 'url' in data || 'name' in data || 'type' in data; const hasLocationFields = 'fbaWarehouse' in data || 'address' in data; return hasDocFields && !hasLocationFields; } transferDocuments().catch(console.error);
