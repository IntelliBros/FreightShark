Setup SMTP Configuration in System Settings

ℹ️ Important: This migration moves SMTP configuration from localStorage to the Supabase database for better security and consistency across sessions.

Instructions:

Go to your Supabase Dashboard
Navigate to SQL Editor (in the left sidebar)
Click "New Query"
Copy the SQL below and paste it into the editor
Update the SMTP values with your actual credentials
Click "Run" to execute the migration

Migration SQL:

-- Setup SMTP Configuration in System Settings -- This stores email configuration in the database instead of localStorage -- First ensure the system_settings table exists CREATE TABLE IF NOT EXISTS system_settings ( key VARCHAR(100) PRIMARY KEY, value TEXT NOT NULL, description TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Insert SMTP settings -- UPDATE THESE VALUES WITH YOUR ACTUAL SMTP CREDENTIALS INSERT INTO system_settings (key, value, description) VALUES ('smtp_host', 'smtp.gmail.com', 'SMTP server hostname'), ('smtp_port', '587', 'SMTP server port'), ('smtp_secure', 'false', 'Use SSL/TLS for SMTP connection'), ('smtp_user', 'your-email@gmail.com', 'SMTP authentication username/email'), ('smtp_pass', 'your-app-password', 'SMTP authentication password'), ('smtp_from_name', 'FreightShark', 'From name for emails'), ('smtp_from_email', 'noreply@freightshark.com', 'From email address'), ('smtp_enabled', 'true', 'Whether SMTP email sending is enabled') ON CONFLICT (key) DO UPDATE SET value = EXCLUDED.value, description = EXCLUDED.description, updated_at = CURRENT_TIMESTAMP; -- Create helper functions CREATE OR REPLACE FUNCTION get_smtp_config() RETURNS JSON AS $$ DECLARE result JSON; BEGIN SELECT json_build_object( 'enabled', (SELECT value FROM system_settings WHERE key = 'smtp_enabled')::boolean, 'host', (SELECT value FROM system_settings WHERE key = 'smtp_host'), 'port', (SELECT value FROM system_settings WHERE key = 'smtp_port')::integer, 'secure', (SELECT value FROM system_settings WHERE key = 'smtp_secure')::boolean, 'auth', json_build_object( 'user', (SELECT value FROM system_settings WHERE key = 'smtp_user'), 'pass', (SELECT value FROM system_settings WHERE key = 'smtp_pass') ), 'from', json_build_object( 'name', (SELECT value FROM system_settings WHERE key = 'smtp_from_name'), 'email', (SELECT value FROM system_settings WHERE key = 'smtp_from_email') ) ) INTO result; RETURN result; END; $$ LANGUAGE plpgsql; -- Grant permissions GRANT EXECUTE ON FUNCTION get_smtp_config() TO authenticated; GRANT EXECUTE ON FUNCTION get_smtp_config() TO anon; GRANT EXECUTE ON FUNCTION get_smtp_config() TO service_role; -- Test the configuration SELECT get_smtp_config(); -- View all SMTP settings SELECT key, value, description FROM system_settings WHERE key LIKE 'smtp_%' ORDER BY key;
